Notes Taking App

Write, save, and delete notes — all stored in your browser with localStorage.

Source Code

// Notes Taking App — localStorage CRUD

let notes = JSON.parse(localStorage.getItem("notes")) || [];

function save() {
  localStorage.setItem("notes", JSON.stringify(notes));
  render();
}

function addNote() {
  const title = document.getElementById("noteTitle").value.trim() || "Untitled";
  const text = document.getElementById("noteText").value.trim();
  if (!text) return;
  notes.push({ id: Date.now(), title, text, date: new Date().toLocaleDateString() });
  document.getElementById("noteTitle").value = "";
  document.getElementById("noteText").value = "";
  save();
}

function deleteNote(id) {
  notes = notes.filter(n => n.id !== id);
  save();
}

function render() {
  const grid = document.getElementById("notesGrid");
  if (!notes.length) {
    grid.innerHTML = `<div class="notes-empty">
      <svg ...></svg><p>No notes yet</p></div>`;
    return;
  }
  grid.innerHTML = notes.map(n => `<div class="note-card">
    <button class="note-del" onclick="deleteNote(${n.id})">✕</button>
    <div class="note-title">${n.title}</div>
    <div class="note-text">${n.text}</div>
    <div class="note-date">${n.date}</div>
  </div>`).join("");
}

render();
