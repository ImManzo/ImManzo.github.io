// Calculator — Vanilla JavaScript

const display = document.getElementById("display");
let current = "0";
let previous = "";
let operation = null;
let shouldReset = false;

function updateDisplay() { display.textContent = current; }

function inputDigit(digit) {
  if (shouldReset) { current = ""; shouldReset = false; }
  if (current === "0" && digit !== ".") { current = digit; }
  else { current += digit; }
  updateDisplay();
}

function inputDecimal() {
  if (shouldReset) { current = "0"; shouldReset = false; }
  if (!current.includes(".")) current += ".";
  updateDisplay();
}

function toggleSign() {
  current = String(parseFloat(current) * -1);
  updateDisplay();
}

function inputPercent() {
  current = String(parseFloat(current) / 100);
  updateDisplay();
}

function handleOperator(op) {
  if (operation && !shouldReset) calculate();
  previous = current;
  operation = op;
  shouldReset = true;
}

function calculate() {
  if (!operation) return;
  const a = parseFloat(previous);
  const b = parseFloat(current);
  let result;
  switch (operation) {
    case "+": result = a + b; break;
    case "-": result = a - b; break;
    case "*": result = a * b; break;
    case "/": result = b === 0 ? "Error" : a / b; break;
    default: return;
  }
  current = String(result);
  operation = null;
  previous = "";
  shouldReset = true;
  updateDisplay();
}

function clearAll() {
  current = "0"; previous = "";
  operation = null; shouldReset = false;
  updateDisplay();
}

document.querySelector(".buttons")
  .addEventListener("click", (e) => {
  const btn = e.target.closest(".btn");
  if (!btn) return;
  const value = btn.dataset.value;
  if (value === "C") return clearAll();
  if (value === "±") return toggleSign();
  if (value === "%") return inputPercent();
  if (value === "=") return calculate();
  if (["+","-","*","/"].includes(value))
    return handleOperator(value);
  if (value === ".") return inputDecimal();
  inputDigit(value);
});
