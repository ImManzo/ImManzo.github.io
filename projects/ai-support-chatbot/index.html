// AI Support Chatbot — FAQ engine + NLP-style matching

const faq = [
  { keywords: ["order", "status", "track"], reply: "Let me check your order status. Could you provide your order number?" },
  { keywords: ["return", "refund", "money back"], reply: "Our return policy is 30 days from delivery. I'll start a return for you right away!" },
  { keywords: ["billing", "invoice", "payment"], reply: "I can help with billing! You can view/download invoices from your account dashboard." },
  { keywords: ["shipping", "delivery", "ship"], reply: "Standard shipping takes 3-5 business days. Express shipping is 1-2 days." },
  { keywords: ["human", "agent", "speak", "representative"], reply: "Let me transfer you to a human agent. Please hold while I connect you..." },
];

function getBotReply(msg) {
  const lower = msg.toLowerCase();
  for (const item of faq) {
    for (const kw of item.keywords) {
      if (lower.includes(kw)) return item.reply;
    }
  }
  return "I'm not sure I understand. Let me transfer you to our support team!";
}
