// AI Resume Analyzer — keyword scoring + smart feedback

const skills = [
  { name: "Frontend", keywords: ["react", "vue", "angular", "html", "css", "javascript", "typescript"] },
  { name: "Backend", keywords: ["node", "express", "django", "flask", "python", "java", "go", "api"] },
  { name: "Database", keywords: ["sql", "mongodb", "postgres", "mysql", "redis"] },
  { name: "Cloud & DevOps", keywords: ["aws", "azure", "docker", "kubernetes", "ci/cd", "devops"] },
  { name: "Soft Skills", keywords: ["lead", "team", "communicat", "manage", "mentor", "agile"] },
];

function analyzeResume() {
  const text = input.value.toLowerCase();
  const results = skills.map(s => {
    const hits = s.keywords.filter(k => text.includes(k)).length;
    return { ...s, score: Math.min(Math.round((hits / s.keywords.length) * 100), 100) };
  });
  displayResults(results);
}
