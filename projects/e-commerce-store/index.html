// E-Commerce Store — product catalog with cart + admin panel

const products = [
  { id: 1, name: "Wireless Headphones", price: 79.99, stock: 12 },
  { id: 2, name: "Mechanical Keyboard", price: 119.99, stock: 8 },
  { id: 3, name: "USB-C Hub", price: 34.99, stock: 25 },
  { id: 4, name: "Portable SSD 1TB", price: 149.99, stock: 5 }
];

let cart = [];
function addToCart(id) {
  const p = products.find(x => x.id === id);
  if (!p || p.stock <= 0) return;
  const item = cart.find(x => x.id === id);
  item ? item.qty++ : cart.push({ ...p, qty: 1 });
  p.stock--;
  render();
}

function switchTab(tab) {
  document.getElementById("storeView").style.display = tab === "store" ? "block" : "none";
  document.getElementById("adminView").style.display = tab === "admin" ? "block" : "none";
}
