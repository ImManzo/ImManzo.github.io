// URL Shortener with localStorage persistence

function shortenUrl() {
  const url = document.getElementById("urlInput").value.trim();
  if (!url) return;

  const code = Math.random().toString(36).substring(2, 7);
  const links = JSON.parse(localStorage.getItem("links")) || [];
  links.push({ code, url, clicks: 0, date: Date.now() });
  localStorage.setItem("links", JSON.stringify(links));

  const shortUrl = `https://short.link/${code}`;
  document.getElementById("shortLink").textContent = shortUrl;
  document.getElementById("shortLink").href = "#";
  document.getElementById("shortResult").style.display = "flex";
  renderLinks();
}
