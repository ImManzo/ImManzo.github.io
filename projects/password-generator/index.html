// Random Password Generator

const upper = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
const lower = "abcdefghijklmnopqrstuvwxyz";
const nums  = "0123456789";
const syms  = "!@#$%^&*()_+-=[]{}|;:,./<>?";

function generatePassword() {
  let chars = "";
  if (document.getElementById("pwUpper").checked) chars += upper;
  if (document.getElementById("pwLower").checked) chars += lower;
  if (document.getElementById("pwNumbers").checked) chars += nums;
  if (document.getElementById("pwSymbols").checked) chars += syms;
  if (!chars) { alert("Select at least one set"); return; }

  const length = parseInt(document.getElementById("pwLength").value);
  let password = "";
  for (let i = 0; i < length; i++) {
    password += chars[Math.floor(Math.random() * chars.length)];
  }
  document.getElementById("pwOutput").value = password;
  updateStrength(length, chars);
}
