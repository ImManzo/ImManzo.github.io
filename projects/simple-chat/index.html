// Simple Chat App — Chatbot with natural responses

const responses = {
  "how are you": "I'm doing great! How was your day? :)",
  "hello": "Hey there! What's up?",
  "hi": "Hi! How can I help you today?",
  "what's your name": "I'm ChatBot, nice to meet you!",
  "bye": "Goodbye! Talk to you later!",
};

function getBotReply(msg) {
  const lower = msg.toLowerCase();
  for (const [key, reply] of Object.entries(responses)) {
    if (lower.includes(key)) return reply;
  }
  return "That's interesting! Tell me more... 😊";
}
